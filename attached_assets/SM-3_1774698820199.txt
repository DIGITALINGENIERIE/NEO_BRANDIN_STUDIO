# INSTRUCTION POUR REPLIT AI

## Module 01.3 — Typography System

### Objectif

Modifier l'application Neo Branding Studio pour implémenter un système de génération de systèmes typographiques **ultra-précis** pour le sous-module Typography System, en utilisant l'exemple parfait fourni comme **golden reference**.

L'objectif est de garantir que **chaque génération** respecte exactement la structure, les spécifications techniques (tailles, graisses, tracking), les URLs Google Fonts, les variables CSS et les classes utilitaires de l'exemple, quels que soient les paramètres d'entrée.

---

## 1. Contexte

**Fichier actuel à modifier** :  
`api-server/src/routes/openai/enhance-prompts.ts` (section Module 01)

**Données d'entrée du formulaire** :
```typescript
interface BrandBrief {
  brandName: string;      // ex: "Theravectys"
  sector: string;         // ex: "tech", "luxury", "streetwear", "fitness", "cosmetic"
  tone: string;           // ex: "professionnel", "audacieux", "minimaliste"
  values: string[];       // ex: ["innovation", "rigueur", "fiabilité"]
  primaryColor: string;   // ex: "#003087" (couleur primaire de la palette)
  secondaryColor: string; // ex: "#6C757D"
  accentColor: string;    // ex: "#2F80ED"
  neutrals: {
    light: string;        // ex: "#F5F7FA"
    dark: string;         // ex: "#0B1220"
    text: string;         // ex: "#101828"
    textSecondary: string;// ex: "#667085"
  };
}
```

**Sortie attendue** : Un prompt structuré qui générera un système typographique complet avec variables CSS, classes utilitaires et paires recommandées.

import { goldenExample } from './golden-example.json';
import { sectorFonts, typographySpecs } from './mapping-rules.json';

interface BrandBrief {
  brandName: string;
  sector: string;
  tone: string;
  values: string[];
  primaryColor: string;
  secondaryColor: string;
  accentColor: string;
  neutrals: {
    light: string;      // fond clair
    dark: string;       // fond sombre
    text: string;       // texte principal
    textSecondary: string; // texte secondaire
  };
}

interface FontConfig {
  name: string;
  url: string;
  weights: string;
  fallbacks: string[];
}

export function buildTypographyPrompt(brief: BrandBrief): string {
  // Récupérer les polices selon le secteur
  const fonts = sectorFonts[brief.sector] || sectorFonts.tech;
  
  // Construire la description des valeurs
  const valuesText = brief.values.join(', ');
  
  // Adapter le style selon le ton
  const toneModifier = getToneModifier(brief.tone);
  
  // Construire le prompt final
  let prompt = goldenExample.goldenExample;
  
  // Remplacer les variables de base
  prompt = prompt.replace(/Theravectys/g, brief.brandName);
  prompt = prompt.replace(/secteur tech/g, `secteur ${brief.sector}`);
  prompt = prompt.replace(/ton professionnel/g, `ton ${brief.tone}`);
  prompt = prompt.replace(/innovation, rigueur scientifique, fiabilité, expertise, engagement/g, valuesText);
  
  // Remplacer les couleurs de contexte
  prompt = prompt.replace(/#0B1220/g, brief.neutrals.dark);
  prompt = prompt.replace(/#F5F7FA/g, brief.neutrals.light);
  prompt = prompt.replace(/#00D1FF/g, brief.accentColor);
  prompt = prompt.replace(/#6C5CE7/g, brief.secondaryColor);
  prompt = prompt.replace(/#101828/g, brief.neutrals.text);
  prompt = prompt.replace(/#667085/g, brief.neutrals.textSecondary);
  
  // Section 1: Police titres
  const headingSection = `
## 1) Police titres (h1/h2/h3)
- Nom de la police: **${fonts.heading.name}** (Google Fonts)
- **Google Fonts URL exacte**: ${fonts.heading.url}
- Graisses à charger: ${fonts.heading.weights}
- Tailles imposées: 
  - h1: ${typographySpecs.heading.h1.size}, line-height ${typographySpecs.heading.h1.lineHeight}, letter-spacing ${typographySpecs.heading.h1.tracking}, font-weight ${typographySpecs.heading.h1.weight}
  - h2: ${typographySpecs.heading.h2.size}, line-height ${typographySpecs.heading.h2.lineHeight}, letter-spacing ${typographySpecs.heading.h2.tracking}, font-weight ${typographySpecs.heading.h2.weight}
  - h3: ${typographySpecs.heading.h3.size}, line-height ${typographySpecs.heading.h3.lineHeight}, letter-spacing ${typographySpecs.heading.h3.tracking}, font-weight ${typographySpecs.heading.h3.weight}
- Usage: titres, sections, hero, navigation principale
- Recommandation typographique: famille ${fonts.heading.name} pour tous les titres, avec une hiérarchie claire basée sur les poids et tailles définis.
`;
  
  // Section 2: Police corps
  const bodySection = `
## 2) Police corps (paragraphes)
- Nom de la police: **${fonts.body.name}** (Google Fonts)
- **URL exacte**: ${fonts.body.url}
- Taille imposée: ${typographySpecs.body.base.size}
- Interligne imposé: ${typographySpecs.body.base.lineHeight}
- Graisses à charger: ${fonts.body.weights}
- Recommandations: 
  - Paragraph spacing: 1.5em
  - Max-width lecture: 70ch pour une lisibilité optimale
  - ${fonts.body.name} assure une excellente lisibilité sur écran comme sur papier
`;
  
  // Section 3: Police accent
  const accentSection = `
## 3) Police accent (CTA/boutons/captions)
- Nom de la police: **${fonts.accent.name}** (Google Fonts)
- **URL exacte**: ${fonts.accent.url}
- Tailles:
  - CTA/button: ${fonts.ctaSize} (choix optimal pour les interfaces tech, offre un bon équilibre entre présence et discrétion)
  - caption/label: ${fonts.captionSize}
- Graisses: 600/700 recommandées
- Recommandations: letter-spacing légèrement positif pour CTA: +0.02em, renforce le caractère "engineered"
- Uppercase: ${fonts.ctaUppercase === 'ON' ? 'ACTIVÉ pour les boutons CTA (renforce l'impact et la lisibilité)' : 'DÉSACTIVÉ (privilégier le capitalisation standard)'}
`;
  
  // Section 4: Fallbacks web-safe
  const fallbacksSection = `
## 4) Fallbacks web-safe (2 par police)
- Police titres: ${fonts.heading.name} → ${fonts.heading.fallbacks.join(', ')}
- Police corps: ${fonts.body.name} → ${fonts.body.fallbacks.join(', ')}
- Police accent: ${fonts.accent.name} → ${fonts.accent.fallbacks.join(', ')}
`;
  
  // Section 5: Variables CSS

\`\`\`css
:root {
  /* Font families */

  --text-2xl: ${typographySpecs.heading.h1.size};

  --leading-tight: ${typographySpecs.heading.h1.lineHeight};
  --leading-normal: ${typographySpecs.body.base.lineHeight};

\`\`\`css
/* Headings */

  font-family: var(--font-heading);
  font-weight: var(--weight-bold);
  font-size: var(--text-2xl);
  line-height: var(--leading-tight);

  color: ${brief.neutrals.text};
}

.heading-lg {
  font-family: var(--font-heading);
  font-weight: var(--weight-semibold);

/* Body */
.body-md {
  font-family: var(--font-body);
  font-weight: var(--weight-regular);
  font-size: var(--text-base);

  font-family: var(--font-body);

  color: ${brief.neutrals.textSecondary};
}

/* CTA Button */
.cta {
  font-family: var(--font-accent);
  font-weight: var(--weight-semibold);
  font-size: ${fonts.ctaSize};
  line-height: 1.2;
  letter-spacing: 0.02em;
  ${fonts.ctaUppercase === 'ON' ? 'text-transform: uppercase;' : ''}
  color: ${brief.neutrals.light};

  padding: 12px 24px;

}

  background: ${adjustColor(brief.primaryColor, -10)};

`;
  
  // Assembler le prompt final
  prompt = prompt.replace(/## 1\) Police titres.*?(?=## 2\) Police corps)/s, headingSection);
  prompt = prompt.replace(/## 2\) Police corps.*?(?=## 3\) Police accent)/s, bodySection);
  prompt = prompt.replace(/## 3\) Police accent.*?(?=## 4\) Fallbacks)/s, accentSection);
  prompt = prompt.replace(/## 4\) Fallbacks web-safe.*?(?=## 5\) Variables CSS)/s, fallbacksSection);
  prompt = prompt.replace(/## 5\) Variables CSS.*?(?=## 6\) Classes utilitaires)/s, cssVariablesSection);
  prompt = prompt.replace(/## 6\) Classes utilitaires.*?(?=## 7\) Paires recommandées)/s, utilityClassesSection);
  prompt = prompt.replace(/## 7\) Paires recommandées.*?(?=---)/s, pairingsSection);
  
  // Ajouter le résumé (Typography rationale)
  const rationale = `
## Typography rationale

**${brief.brandName}** adopte un système typographique ${brief.tone} aligné sur son positionnement ${brief.sector}.

- **${fonts.heading.name}** pour les titres: ${getFontRationale(fonts.heading.name, brief.sector)}
- **${fonts.body.name}** pour le corps: lisibilité optimale sur tous supports
- **${fonts.accent.name}** pour l'accent: renforce le caractère ${brief.sector === 'tech' ? 'technique et précis' : brief.sector === 'luxury' ? 'élégant et distinctif' : 'identitaire et mémorable'}

Cette combinaison garantit:
- Hiérarchie visuelle claire (h1→h2→h3→body)
- Accessibilité WCAG 2.1 AA (contrasts validés)
- Scalabilité responsive (16px base)
- Cohérence cross-platform (fallbacks web-safe)
`;
  
  // Insérer le résumé au début
  prompt = prompt.replace(/Génère le système typographique/, `${rationale}\n\nGénère le système typographique`);
  
  return prompt;
}

// Fonctions utilitaires
function getToneModifier(tone: string): string {
  const modifiers: Record<string, string> = {
    professionnel: " avec une approche maîtrisée et experte",
    audacieux: " avec une touche d'innovation disruptive",
    minimaliste: " dans une approche épurée et essentielle"
  };
  return modifiers[tone] || "";
}

function getFontRationale(fontName: string, sector: string): string {
  const rationales: Record<string, string> = {
    'Space Grotesk': 'géométrie technique, idéale pour l\'univers tech',
    'Inter': 'lisibilité supérieure, standard du design system moderne',
    'JetBrains Mono': 'précision engineering, parfait pour les données',
    'Cormorant Garamond': 'élégance intemporelle, héritage typographique',
    'Montserrat': 'modernité géométrique, polyvalence premium',
    'Bebas Neue': 'impact visuel maximal, identité streetwear forte',
    'Poppins': 'dynamisme géométrique, énergie contrôlée',
    'Open Sans': 'neutralité fonctionnelle, performance sportive',
    'Playfair Display': 'élégance naturelle, sophistication cosmétique',
    'Nunito Sans': 'douceur fonctionnelle, bien-être visuel'
  };
  return rationales[fontName] || `parfaite pour l'univers ${sector}`;
}

function adjustColor(hex: string, percent: number): string {
  // Conversion simplifiée pour le prompt
  const r = parseInt(hex.slice(1, 3), 16);
  const g = parseInt(hex.slice(3, 5), 16);
  const b = parseInt(hex.slice(5, 7), 16);
  
  const adjust = (c: number) => {
    const newC = c + (c * percent / 100);
    return Math.min(255, Math.max(0, Math.round(newC)));
  };
  
  const newR = adjust(r);
  const newG = adjust(g);
  const newB = adjust(b);
  
  return `#${newR.toString(16).padStart(2, '0')}${newG.toString(16).padStart(2, '0')}${newB.toString(16).padStart(2, '0')}`;
}
