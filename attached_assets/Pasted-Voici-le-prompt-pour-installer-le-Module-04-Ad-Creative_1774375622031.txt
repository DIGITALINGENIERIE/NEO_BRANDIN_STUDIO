def get_meta_ad_prompt(format_type: str, style: str, sector: str, cta: str, colors: str) -> str:
    """Génère un prompt pour Meta Ads"""
    prompt = META_ADS_PROMPTS.get(format_type, META_ADS_PROMPTS["feed_image"])
    return prompt.format(style=style, sector=sector, cta=cta, colors=colors)

def get_google_display_prompt(format_type: str, style: str, sector: str, colors: str) -> str:
    """Génère un prompt pour Google Display"""

from typing import Dict, Any, List
from modules.base_module import BaseModule
from modules.module_04_ad_creatives.prompts import (
    get_meta_ad_prompt,
    get_google_display_prompt,
    get_tiktok_ad_prompt,
    get_carousel_slide_prompt,
    get_ad_copy_template
)

class AdCreativesOrchestrator(BaseModule):
    """
    MODULE 04 — AD CREATIVES
    Génère tous les prompts pour créer les publicités d'une marque
    """
    
    def __init__(self):
        super().__init__("AdCreatives")
    
    def generate(self, brief: Dict[str, Any]) -> Dict[str, Any]:
        """
        Génère tous les prompts du Module 04
        
        Input brief attendu:
        {
            "brand_name": "LUXEOR",
            "sector": "bijou",
            "product_name": "Montre Élégance",
            "product_description": "...",
            "benefits": ["élégance", "précision", "héritage"],
            "target_audience": "femmes_35_50",
            "colors": ["or rose", "noir"],
            "style": "luxe",
            "promo_code": "LUXEOR20",
            "discount": 20,
            "duration_days": 7,
            "free_shipping": 100,
            "stock": 50,
            "problem": "bijoux qui ternissent"
        }
        """
        
        # Extraire les données
        brand_name = brief.get("brand_name", "Ma Marque")
        sector = brief.get("sector", "general")
        product_name = brief.get("product_name", "notre produit")
        benefits = brief.get("benefits", ["qualité", "performance"])
        target_audience = brief.get("target_audience", "mixte")
        colors = brief.get("colors", ["or", "noir"])
        style = brief.get("style", self._detect_style(sector))
        promo_code = brief.get("promo_code", brand_name.upper()[:4])
        discount = brief.get("discount", 20)
        duration_days = brief.get("duration_days", 7)
        free_shipping = brief.get("free_shipping", 100)
        stock = brief.get("stock", 50)
        problem = brief.get("problem", "problèmes quotidiens")
        
        # Variables pour personnalisation des copy
        copy_context = {
            "product": product_name,
            "brand": brand_name,
            "sector": sector,
            "benefit_1": benefits[0] if benefits else "qualité",
            "benefit_2": benefits[1] if len(benefits) > 1 else "performance",
            "benefit_short": benefits[0][:20] if benefits else "qualité",
            "code": promo_code,
            "discount": discount,
            "duration": duration_days,
            "free_shipping": free_shipping,
            "days": "30",
            "stock": stock,
            "problem": problem
        }
        
        self._log(f"Génération des prompts publicitaires pour {brand_name} ({sector})")
        
        # 1. Meta Ads (feed image, feed video, stories, reels)
        meta_ads = {}
        meta_formats = ["feed_image", "feed_video", "stories", "reels"]
        
        for fmt in meta_formats:
            prompt = get_meta_ad_prompt(fmt, style, sector, "Acheter maintenant", ", ".join(colors))
            personalized = self._personalize_prompt(prompt, copy_context)
            enhanced = self._enhance_prompt(personalized, copy_context)
            
            meta_ads[fmt] = {
                "prompt": enhanced,
                "parameters": {
                    "format": fmt,
                    "dimensions": "1080x1080" if fmt == "feed_image" or fmt == "feed_video" else "1080x1920",
                    "cta": self._get_cta_for_format(fmt)
                }
            }
        
        # 2. Google Display (6 formats)
        google_display = {}
        google_formats = ["leaderboard", "medium_rectangle", "large_rectangle", "mobile_banner", "half_page", "billboard"]
        
        for fmt in google_formats:
            prompt = get_google_display_prompt(fmt, style, sector, ", ".join(colors))
            personalized = self._personalize_prompt(prompt, copy_context)
            enhanced = self._enhance_prompt(personalized, copy_context)
            
            google_display[fmt] = {
                "prompt": enhanced,
                "parameters": {
                    "format": fmt,
                    "dimensions": self._get_google_dimensions(fmt),
                    "html5_available": True
                }
            }
        
        # 3. TikTok Ads
        tiktok_prompt = get_tiktok_ad_prompt(style, sector, product_name, brand_name)
        tiktok_personalized = self._personalize_prompt(tiktok_prompt, copy_context)
        tiktok_enhanced = self._enhance_prompt(tiktok_personalized, copy_context)
        
        tiktok_ads = {
            "prompt": tiktok_enhanced,
            "parameters": {
                "format": "9:16",
                "dimensions": "1080x1920",
                "duration": "15-30s",
                "sound_required": True
            }
        }
        
        # 4. Carousel Ads (5 slides)
        carousel = {}
        slide_types = ["hook", "problem", "solution", "proof", "cta"]
        
        for i, slide_type in enumerate(slide_types, 1):
            prompt = get_carousel_slide_prompt(i, slide_type, style)
            personalized = self._personalize_prompt(prompt, copy_context)
            enhanced = self._enhance_prompt(personalized, copy_context)
            
            carousel[f"slide_{i}"] = {
                "prompt": enhanced,
                "role": slide_type,
                "parameters": {
                    "dimensions": "1080x1080",
                    "format": "image"
                }
            }
        
        # 5. Ad Copy (4 variantes par plateforme)
        ad_copy = {}
        platforms = ["meta_feed", "meta_stories", "google_display", "tiktok"]
        
        for platform in platforms:
            template = get_ad_copy_template(platform, sector)
            copy_variants = {}
            
            for field, variants in template.items():
                personalized_variants = []
                for variant in variants[:4]:  # 4 variantes max
                    personalized = self._personalize_copy(variant, copy_context)
                    personalized_variants.append(personalized)
                copy_variants[field] = personalized_variants
            
            ad_copy[platform] = copy_variants
        
        # 6. Performance Predictor (recommandations)
        performance = {
            "ctr_factors": {
                "facteurs": [
                    {"factor": "Présence de visage", "impact": "+30% CTR"},
                    {"factor": "Texte overlay <20%", "impact": "+15% CTR"},
                    {"factor": "Couleurs contrastées", "impact": "+10% CTR"},
                    {"factor": "CTA visible sans scroll", "impact": "+20% CTR"},
                    {"factor": "Urgence (stock/offre)", "impact": "+25% CTR"},
                    {"factor": "Preuve sociale", "impact": "+18% CTR"}
                ]
            },
            "recommendations": [
                "Tester 2-3 variantes de créatives simultanément",
                "Garder le même CTA sur toutes les variantes",
                "Tester sur 3-5 jours minimum",
                "Analyser: CTR, CPC, ROAS",
                "Garder la variante gagnante, tester contre une nouvelle"
            ],
            "metrics_to_track": ["CTR", "CPC", "ROAS", "Conversion Rate", "Cost per Conversion"]
        }
        
        # Retourner tous les prompts
        return {
            "ad_creatives": {
                "meta_ads": {
                    "agent": "AI Poster Agent / Creative Ad Video Agent",
                    "creatives": meta_ads,
                    "parameters": {
                        "platforms": ["Facebook", "Instagram"],
                        "formats": meta_formats,
                        "variants_per_format": 1
                    }
                },
                "google_display": {
                    "agent": "AI Poster Agent / Brand Design Agent",
                    "banners": google_display,
                    "parameters": {
                        "formats": google_formats,
                        "html5_animated": True,
                        "dimensions": self._get_all_google_dimensions()
                    }
                },
                "tiktok_ads": {
                    "agent": "Creative Ad Video Agent",
                    "creatives": tiktok_ads,
                    "parameters": {
                        "format": "9:16",
                        "duration": "15-30s",
                        "sound_trending": True,
                        "effects_trending": True
                    }
                },
                "carousel_ads": {
                    "agent": "AI Poster Agent",
                    "carousel": carousel,
                    "parameters": {
                        "slides": 5,
                        "narrative": ["hook", "problem", "solution", "proof", "cta"],
                        "dimensions": "1080x1080"
                    }
                },
                "ad_copy": {
                    "agent": "Manual (copywriting)",
                    "variants": ad_copy,
                    "parameters": {
                        "platforms": platforms,
                        "variants_per_platform": 4,
                        "characters_optimized": True
                    }
                },
                "performance_predictor": {
                    "analysis": performance,
                    "recommendations": [
                        "Tester d'abord Meta Feed Image et Stories",
                        "Google Display: prioriser Medium Rectangle et Mobile Banner",
                        "TikTok: tester 3 versions avec des sons différents",
                        "Carousel: tester narrative 'problème-solution' vs 'storytelling'"
                    ]
                }
            },
            "metadata": {
                "module": "04_ad_creatives",
                "brand_name": brand_name,
                "sector": sector,
                "meta_formats": len(meta_formats),
                "google_formats": len(google_formats),
                "carousel_slides": 5,
                "copy_variants": 4,
                "llm_enhanced": bool(self._enhance_prompt("test", {}) != "test")
            }
        }
    
    def _personalize_prompt(self, prompt: str, context: Dict) -> str:
        """Personnalise un prompt avec le contexte"""
        for key, value in context.items():
            prompt = prompt.replace(f"{{{key}}}", str(value))
        return prompt
    
    def _personalize_copy(self, text: str, context: Dict) -> str:
        """Personnalise un texte publicitaire"""
        for key, value in context.items():
            text = text.replace(f"{{{key}}}", str(value))
        return text
    
    def _detect_style(self, sector: str) -> str:
        """Détecte le style publicitaire selon le secteur"""
        mapping = {
            "bijou": "luxe, élégant, raffiné",
            "luxe": "premium, exclusif, intemporel",
            "cosmétique": "frais, naturel, lumineux",
            "skincare": "doux, naturel, apaisant",
            "tech": "moderne, dynamique, futuriste",
            "fitness": "énergique, motivant, dynamique",
            "mode": "tendance, moderne, élégant",
            "streetwear": "urban, dynamique, cool"
        }
        return mapping.get(sector, "moderne, professionnel")
    
    def _get_cta_for_format(self, format_type: str) -> str:
        """Retourne le CTA recommandé pour un format"""
        mapping = {
            "feed_image": "Acheter maintenant",
            "feed_video": "En savoir plus",
            "stories": "Swipe up",
            "reels": "Lien en bio"
        }
        return mapping.get(format_type, "Découvrir")
    
    def _get_google_dimensions(self, format_type: str) -> str:
        """Retourne les dimensions pour un format Google"""
        mapping = {
            "leaderboard": "728x90",
            "medium_rectangle": "300x250",
            "large_rectangle": "336x280",
            "mobile_banner": "320x100",
            "half_page": "300x600",
            "billboard": "970x250"
        }
        return mapping.get(format_type, "300x250")
    
    def _get_all_google_dimensions(self) -> List[str]:
        """Retourne toutes les dimensions Google"""
        return ["728x90", "300x250", "336x280", "320x100", "300x600", "970x250"]
