# PROMPT — INSTALLATION MODULE 06 — COPY & CONTENT

```
Ajoute le MODULE 06 — COPY & CONTENT au projet existant.

Le Module 06 génère des prompts optimisés pour créer tout le contenu textuel d'une marque: fiche produit complète (titre, description, bullet points, FAQ), captions pour chaque visuel, hashtags optimisés par plateforme, séquence emails (3), et 10 reviews clients réalistes.

Le module doit s'intégrer avec les Modules 01, 02, 03, 04 et 05 déjà installés.

## STRUCTURE À AJOUTER

Ajoute ce dossier et ces fichiers:

```
modules/module_06_copy_content/
├── __init__.py
├── orchestrator.py
└── prompts.py
```

def get_product_sheet_template(sector: str) -> dict:
    """Retourne le template de fiche produit pour un secteur"""
    return PRODUCT_SHEET_TEMPLATES.get(sector, PRODUCT_SHEET_TEMPLATES["bijou"])

def get_caption(platform: str, variant: int, context: dict) -> str:
    """Retourne une caption personnalisée"""
    templates = CAPTION_TEMPLATES.get(platform, CAPTION_TEMPLATES["instagram_feed"])
    template = templates[variant % len(templates)]
    for key, value in context.items():
        template = template.replace(f"{{{key}}}", str(value))
    return template

def get_hashtags(sector: str, brand: str, product: str) -> list:
    """Retourne une liste de hashtags optimisés"""
    hashtag_set = HASHTAG_SETS.get(sector, HASHTAG_SETS["bijou"])
    all_hashtags = []
    all_hashtags.extend(hashtag_set["general"])
    all_hashtags.extend(hashtag_set["specific"])
    all_hashtags.extend(hashtag_set["trending"])
    all_hashtags.extend([h.replace("{brand}", brand).replace("{product}", product) for h in hashtag_set["brand"]])
    return all_hashtags[:20]  # Limite à 20 hashtags

def get_email_sequence(email_type: str, context: dict) -> dict:
    """Retourne une séquence email personnalisée"""
    template = EMAIL_SEQUENCES.get(email_type, EMAIL_SEQUENCES["launch"])
    subject = template["subject"]
    body = template["body"]
    for key, value in context.items():
        subject = subject.replace(f"{{{key}}}", str(value))
        body = body.replace(f"{{{key}}}", str(value))
    return {"subject": subject, "body": body}

def get_reviews(product_name: str, brand: str) -> list:
    """Retourne 10 reviews clients réalistes"""
    reviews = []
    for template in REVIEW_TEMPLATES:
        review = template.copy()
        review["content"] = review["content"].replace("{product}", product_name).replace("{brand}", brand)
        reviews.append(review)
    return reviews

class CopyContentOrchestrator(BaseModule):
    """
    MODULE 06 — COPY & CONTENT
    Génère tous les prompts pour créer le contenu textuel d'une marque
    """
    
    def __init__(self):
        super().__init__("CopyContent")
    
    def generate(self, brief: Dict[str, Any]) -> Dict[str, Any]:
        """
        Génère tous les prompts du Module 06
        
        Input brief attendu:
        {
            "brand_name": "LUXEOR",

            "warranty": 2,
            "stock": 50,
            "first_name": "Client"  # pour emails
        }
        """
        
        # Extraire les données
        brand_name = brief.get("brand_name", "Ma Marque")
        sector = brief.get("sector", "general")
        product_name = brief.get("product_name", "notre produit")
        product_description = brief.get("product_description", "")
        product_features = brief.get("product_features", [])
        benefits = brief.get("benefits", ["qualité", "performance"])
        target_audience = brief.get("target_audience", "mixte")
        discount = brief.get("discount", 20)
        promo_code = brief.get("promo_code", brand_name.upper()[:4])
        free_shipping = brief.get("free_shipping", 100)
        delivery_days = brief.get("delivery_days", 3)
        return_days = brief.get("return_days", 30)
        warranty = brief.get("warranty", 2)
        stock = brief.get("stock", 50)
        first_name = brief.get("first_name", "Client")
        
        # Contexte pour personnalisation
        context = {
            "product": product_name,
            "brand": brand_name,
            "sector": sector,
            "benefit": benefits[0] if benefits else "qualité",
            "benefit_1": benefits[0] if benefits else "qualité",
            "benefit_2": benefits[1] if len(benefits) > 1 else "performance",
            "material": product_features[0] if product_features else "matériaux d'exception",
            "feature": product_features[1] if len(product_features) > 1 else "finitions soignées",
            "movement": product_features[2] if len(product_features) > 2 else "précision",
